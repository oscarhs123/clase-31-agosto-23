# Comandos Bash:

mkdir <nombre_directorio> - crea un directorio nuevo
cd <directorio_destino> - Cambia de directorio al directorio destino
pwd - Muestra la ruta del directorio actual
mkdir -crea un nuevo directorio
ls ista el contenido del directorio
rm <archivo> - Elimina un archivo
rm -r <directorio> - Elimina un directorio

touch crea o actualiza un archivo
cat "reproduce" lo que hay en un archivo
